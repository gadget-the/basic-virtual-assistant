def getGPE2(text, lst = False):
    """ new attempt at making a function that extracts locations from the input """
    if text[-1] not in ["?", ".", "!"]:
        text += "?"

    chunked = ne_chunk(pos_tag(word_tokenize(text)))
    NER = spacy.load("en_core_web_sm")
    tagged = NER(text)

    nameEnts = []

    # use spacy to check for 'GPE'-type named entities
    for word in tagged.ents:
        if str(word.label_) == 'GPE':
            nameEnts.append(word.text)
    
    # use nltk to check for 'GPE'-type named entities
    for subtree in chunked:
        if type(subtree) == Tree and subtree.label() == 'GPE':
            nameEnts.extend([t[0] for t in subtree.leaves()])

    # more in-depth way of removing duplicates, while preserving the order of the list
    nameEnts.reverse()
    for i, x in enumerate(nameEnts):
        m = list(nameEnts)
        m.remove(x)
        if any(x.lower() in word_tokenize(y.lower()) for y in m):
            nameEnts[i] = ""
    nameEnts.reverse()
    nameEnts = [i for i in nameEnts if i]

    if nameEnts:
        if lst:
            return nameEnts
        else:
            return ", ".join(nameEnts)
    else:
        return None

def timExtr(inp = ""):
    ''' outputs a dictionary of time related info based on the input '''
    # inp = str(inp.lower)
    inp = mathParse.parser(inp.lower())
    words = word_tokenize(inp)
    ofDay = ["am", "pm", "a.m", "p.m", "a.m.", "p.m.", "o'clock", "oclock"]
    timeWords = ["yesterday", "today", "tomorrow", "sunday", "monday", "tuesday", "wednesday", "thursday", "friday", "saturday", "next week", "this week"]
    units = ["year", "month", "week", "day", "hour", "minute", "second"]
    months = ["january", "february", "march", "april", "may", "june", "july", "august", "september", "october", "november", "december"]
    timingSet = False
    timing = {}

    # check for a time of the day(i.e. "8 a.m", "5 o'clock")
    for d in ofDay:
        if d in inp:
            for i, n in enumerate(words):
                if d == n:
                    timingSet = True
                    timing['elapsed'] = False # specify that it is a time at which to do the thing
                    if words[i - 1].isdigit():
                        if "p" in d: # convert to 24-hour clock
                            timing['oClock'] = int(words[i - 1]) + 12
                        else:
                            timing['oClock'] = int(words[i - 1])
                    break

    # if not timing['oClock']:
    #     for i in words:
    #         if ":" in i:
    #             timing['oClock']

    # check for a day/day range?(i.e. "tomorrow", "today", "next week")
    for t in timeWords:
        if t in inp:
            for i, n in enumerate(words):
                if t == n:
                    timingSet = True
                    timing['elapsed'] = False
                    timing['time'] = t
                    break

    # check for specified date
    # works for dates like 27 of December, 2018 and December 27th, 2018. need to make it work for 12/27/2018 and 27/12/2018
    for m in months:
        if m in inp:
            for i, n in enumerate(words):
                if m == n:
                    timingSet = True
                    timing['elapsed'] = False
                    date = ""

                    if words[i - 1].isdigit(): # if day is first: 1st 
                        date += words[i - 1]
                    elif words[i - 1] == "of":
                        if words[i - 2][0].isdigit():
                            date += words[i - 2] + " of "
                        # elif words[i - 2] == "first":
                        #     date += "1st"
                        # elif words[i - 2] == "second":
                        #     date += "2nd"
                        # elif words[i - 2] == "third":
                        #     date += "3rd"
                        # elif words[i - 2] == "fifth":
                        #     date += "5th"
                        # else:
                        #     p = mathParse.parser(words[i - 2].replace())

                    date += m

                    if i < len(words) - 1:
                        if words[i + 1].isdigit():
                            date += " " +words[i + 1]
                        elif words[i + 1][0].isdigit():
                            date +=" " + words[i + 1]

                    if i < len(words) - 2:
                        if words[i + 2].isdigit(): # might cause some false positives
                            date += " " + words[i + 2]
                        
                    timing['date'] = date
                    break
    
    # if timing is still not set, check for a specific time to pass
    if not timing:
        timing['elapsed'] = True # specify that it is an amount of time that must elapse before the thing is to happen
        for u in units:
            if u in inp:
                for i, n in enumerate(words):
                    if u in n:
                        timingSet = True
                        if words[i - 1].isdigit():
                            timing[u + "s"] = int(words[i - 1])
                        elif words[i - 1] == "an" or words[i - 1] == "a":
                            timing[u + "s"] = 1
                        else:
                            timing[u + "s"] = 0
            else:
                timing[u + "s"] = 0
            
    # return timing
    return timing if timingSet else None # probably not a good way to do this

def lookUp(n = "", t = "file"):
    with open('lookUp.json', 'r') as fp:
        fileRef = json.load(fp)

    for p in fileRef[t]:
        if p in n.lower():
            return p
        else:
            for cN in fileRef[t][p]:
                if cN in n.lower():
                    return p

    return None
